}

class Player {
  int playerHP = 100;
  
   int getHP() {
    return playerHP;
  }
  
  void takeDamage(int damage) {
    playerHP -= damage;
  }

  bool isDead() {
    return playerHP <= 0;
  }
}

class Enemy {
  int enemyHP = 100;
  
    int getHP() {
    return enemyHP;
  }

  void takeDamage(int damage) {
    enemyHP -= damage;
  }

  bool isDestroyed() {
    return enemyHP <= 0;
  }
}

void main() {
  SpaceBattle spaceBattle = SpaceBattle();
  spaceBattle.startBattle();
}
